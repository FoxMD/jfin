     * Returns the position of the wanted column.
     * @param what Wanted column
     * @return Position of wanted column
     */
    public static int getColumnPosition(String what) {
        int len = TABLE_HEADER.length;
        int i = 0;

        while (i < len) {
            if (TABLE_HEADER[i] == what) {
                return i;
            } else {
                i = i + 1;
            }
        }
        return -1;
    }
